def swara_frequency(base_sa: float, swara: str, octave: int = 0) -> float:
    """
    Calculate frequency of any swara
    
    > "‡§∏‡§™‡•ç‡§§ ‡§∏‡•ç‡§µ‡§∞‡§æ‡§É ‡§™‡•ç‡§∞‡§•‡§Æ‡§æ‡§¶‡§Ø‡§É"

    > ‚Äî Natyashastra
    """
    ratios = {

        "Re": 9/8,
        "Ga_komal": 32/27,
        "Ga": 5/4,
        "Ma": 4/3,

        "Ni_komal": 16/9,
        "Ni": 15/8,

    ratio = ratios.get(swara, 1)

def raga_time(hour: int) -> list:
    """
    Suggest appropriate ragas for time of day
    
    > "‡§∞‡§æ‡§ó ‡§ï‡§æ‡§≤‡§æ‡§®‡•Å‡§∏‡§æ‡§∞‡§ø‡§£‡§É"
    > "RƒÅga kƒÅlƒÅnusƒÅri·πáa·∏•"

        (9, 12): ["Asavari", "Todi", "Bairagi"],
        (12, 15): ["Sarang", "Madhuvanti", "Brindavani Sarang"],
        (15, 18): ["Bhimpalasi", "Multani", "Patdeep"],
        (18, 21): ["Yaman", "Kalyan", "Shankara"],

        (0, 3): ["Lalit", "Sohni", "Bhatiyar"],
        (3, 6): ["Bhairavi", "Sindhu Bhairavi"]

    for (start, end), ragas in time_ragas.items():
