```python
def apply_svara_sandhi(first: str, second: str) -> str:
    """
    Apply Svara (vowel) Sandhi rules
    
    > "‡§Ö‡§ï‡§É ‡§∏‡§µ‡§∞‡•ç‡§£‡•á ‡§¶‡•Ä‡§∞‡•ç‡§ò‡§É"
    > "Aka·∏• savar·πáe dƒ´rgha·∏•"
    > "A followed by similar vowel becomes long."
    > ‚Äî Ashtadhyayi 6.1.101
    """
    # Simplified for demonstration
    savarna_pairs = {
        ('a', 'a'): 'ƒÅ', ('ƒÅ', 'a'): 'ƒÅ', ('a', 'ƒÅ'): 'ƒÅ', ('ƒÅ', 'ƒÅ'): 'ƒÅ',
        ('i', 'i'): 'ƒ´', ('ƒ´', 'i'): 'ƒ´', ('i', 'ƒ´'): 'ƒ´', ('ƒ´', 'ƒ´'): 'ƒ´',
        ('u', 'u'): '≈´', ('≈´', 'u'): '≈´', ('u', '≈´'): '≈´', ('≈´', '≈´'): '≈´',
    }
    
    guna_rules = {
        ('a', 'i'): 'e', ('a', 'ƒ´'): 'e', ('ƒÅ', 'i'): 'e', ('ƒÅ', 'ƒ´'): 'e',
        ('a', 'u'): 'o', ('a', '≈´'): 'o', ('ƒÅ', 'u'): 'o', ('ƒÅ', '≈´'): 'o',
    }
    
    last_of_first = first[-1] if first else ''
    first_of_second = second[0] if second else ''
    
    key = (last_of_first, first_of_second)
    
    if key in savarna_pairs:
        return first[:-1] + savarna_pairs[key] + second[1:]
    elif key in guna_rules:
        return first[:-1] + guna_rules[key] + second[1:]
    else:
        return first + second  # No sandhi
```
